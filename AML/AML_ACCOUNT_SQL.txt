-- 대출 및 투자관련 --------------------------------------------------------------

SELECT
	SYSTEM_DIV,ACCOUNT_NO,
	PROD_CD,PROD_NM,
	REG_USER_ID,REG_DT,LAST_CHANGE_USER_ID,LAST_CHANGE_DT,CREATE_DT
FROM
	AML_PROD_BASE
ORDER BY
	LAST_CHANGE_DT DESC,
	REG_DT DESC;

SELECT
	SYSTEM_DIV,ACCOUNT_NO,
	CUSTOMER_NO,
	ACCOUNT_STS_CD,ACCOUNT_ORG_DIV,CURRENCY_CD,
	PROD_CD,PROD_RA_CD,PROD_REP_CD,ACCOUNT_OPEN_DD,CLOSE_DD,
	ACCOUNT_OPEN_PURPOSE_CD,ACCOUNT_OPEN_PURPOSE_NM,ACCOUNT_DIV,
	TOT_PAYMENT_AMT,LOAN_NEW_DD,LOAN_END_DD,LOAN_NEW_AMT,
	REG_USER_ID,REG_DT,LAST_CHANGE_USER_ID,LAST_CHANGE_DT,CREATE_DT
FROM
	AML_ACCOUNT_BASE
ORDER BY
	LAST_CHANGE_DT DESC,
	REG_DT DESC;

SELECT
	SYSTEM_DIV,ACCOUNT_NO,PROD_CD,TRAN_DD,TRAN_SEQ_NO,
	CUSTOMER_NO,
	TRAN_TIME,TRAN_CHANNEl_CD,TRAN_CHANNEl_NM,TRAN_WAY_CD,TRAN_WAY_NM,TRAN_KIND_CD,TRAN_KIND_NM,
	SUMMARY_CD,SUMMARY_NM,
	STR_TRAN_KIND_CD,STR_TRAN_KIND_NM,
	RLT_FINAN_ORG_COUNTRY_CD,RLT_FINAN_ORG_CD,RLT_FINAN_ORG_NM,RLTACC_NO,RLTACC_OWNER_NM,
	ACCOUNT_TP_DIV,ACCOUNT_DIV_TP_NM,
	CURRENCY_CD,TRAN_AMT,WON_TRAN_AMT,FEX_TRAN_AMT,USD_EXCHANGE_AMT,TRAN_PURPOSE_CD,TRAN_PURPOSE_NM,
	REG_USER_ID,REG_DT,LAST_CHANGE_USER_ID,LAST_CHANGE_DT,CREATE_DT
FROM
	AML_ACCOUNT_TRAN_BASE
ORDER BY
	LAST_CHANGE_DT DESC,
	REG_DT DESC;
